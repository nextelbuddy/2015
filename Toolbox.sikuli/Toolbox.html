def Navigatepage(pPagecount,pForward):
    if pForward=='True':
        for x in range(0, pPagecount):
            type(Key.F6)
    else:
        for x in range(0, pPagecount):
            keyDown(Key.SHIFT)
            type(Key.F6)
            keyUp(Key.SHIFT)


def Navigatelist(pListcount,pDown):
    if pDown=='True':
        for x in range(0, pListcount):
            type(Key.DOWN)
    else:
        for x in range(0, pListcount):
            type(Key.UP)

    type('f', KeyModifier.ALT)
    type('b')
    wait(Pattern().similar(0.91), 30)
    type(Key.ENTER)
    wait(, 30)
    #('- Saving file path and name')
    wait(.5)
    paste(pDBLocation)
    wait(1)
    type('s', KeyModifier.ALT)
    if exists():
        type('y', KeyModifier.ALT)
    wait(Pattern().similar(0.89), 120)
    type(Key.ENTER)
    waitVanish(Pattern().similar(0.89))
